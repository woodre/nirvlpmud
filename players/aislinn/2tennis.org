id(str) {
  return str == "gate";
}
 
open(str) {
  if(str && str != "gate")
    return 0;
  if(door_is_open)
    return 0;
door_is_open = 1;
write("Ok.\n");
say(call_other(this_player(), "query_name") + " opened the gate.\n");
return 1;
}
 
north() {
  if(!door_is_open) {
  write("The gate is closed.\n");
  return 1;
}
 
if(gguard && present (gguard,this_object())) {
  write("The guard says 'Members only!'\n");
  return 1;
}
