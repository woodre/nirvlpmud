add_cmds() 
  {
   int x; string cmd;
   cmds = get_dir(CMD_DIR);
   for(x=0;x<sizeof(cmds);x++)
   if(sscanf(cmds[x],"%s.c",cmd)) 
     {
     if(cmd == "\"") 
       { 
       add_action("cmd_hook"); add_xverb(cmd); 
       }
     else if(cmd == "'") 
       { 
       add_action("cmd_hook"); add_xverb(cmd); 
       }
     else add_action("cmd_hook",cmd); 
     }
   add_action("cmd_emote"); add_xverb(":");
  }
